This is a post to check how Code Block formatting works

# include<stdio.h>
int main (int argc, char **argv) {
    printf("Hello, world
");
    printf("%i
", argc);
    for (int i = 0; argv[i] != NULL; ++i) {
        printf("%s
", argv[i]);
    }
    return 0;
}
